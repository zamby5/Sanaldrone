addEventListener('fetch', event => {
  event.respondWith(handleRequest(event.request));
});

const TKGM = {
  il:      'https://parselsorgu.tkgm.gov.tr/app/modules/administrativeQuery/data/ilListe.json',
  ilce:    'https://cbsapi.tkgm.gov.tr/megsiswebapi.v3.1/api/idariYapi/ilceListe',
  mahalle: 'https://cbsapi.tkgm.gov.tr/megsiswebapi.v3.1/api/idariYapi/mahalleListe',
  parsel:  'https://cbsapi.tkgm.gov.tr/megsiswebapi.v3.1/api/parsel',
};

const REQ_H = {
  'Referer': 'https://parselsorgu.tkgm.gov.tr/',
  'Origin': 'https://parselsorgu.tkgm.gov.tr',
  'User-Agent': 'Mozilla/5.0 Chrome/120.0.0.0',
  'Accept': 'application/json',
};

const RES_H = {
  'Access-Control-Allow-Origin': '*',
  'Content-Type': 'application/json;charset=utf-8',
  'Cache-Control': 'public, max-age=300',
};

async function handleRequest(req) {
  if (req.method === 'OPTIONS') return new Response(null, {status:204, headers:RES_H});
  const parts = new URL(req.url).pathname.replace(/^\/+/,'').split('/');
  let url = null;
  if (parts[0]==='il') url = TKGM.il;
  else if (parts[0]==='ilce' && parts[1]) url = TKGM.ilce+'/'+parts[1];
  else if (parts[0]==='mahalle' && parts[1]) url = TKGM.mahalle+'/'+parts[1];
  else if (parts[0]==='parsel' && parts[1] && parts[2] && parts[3]) url = TKGM.parsel+'/'+parts[1]+'/'+parts[2]+'/'+parts[3];
  else return new Response(JSON.stringify({error:'gecersiz'}),{status:400,headers:RES_H});
  try {
    const r = await fetch(url, {headers:REQ_H});
    return new Response(await r.text(), {status:r.ok?200:r.status, headers:RES_H});
  } catch(e) {
    return new Response(JSON.stringify({error:e.message}),{status:502,headers:RES_H});
  }
}
